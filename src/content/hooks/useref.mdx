# useRef

`useRef` returns a **mutable ref object** whose `.current` property is initialized with the given value. The key difference from state: **changing `.current` does NOT cause a re-render**.

```tsx
const ref = useRef(initialValue)
// ref.current === initialValue
```

## Two Primary Use Cases

### 1. Accessing DOM nodes

```tsx
function AutoFocusInput() {
  const inputRef = useRef<HTMLInputElement>(null)

  useEffect(() => {
    inputRef.current?.focus()   // directly access the DOM element
  }, [])

  return <input ref={inputRef} placeholder="I auto-focus on mount" />
}
```

### 2. Storing mutable values without re-renders

```tsx
function StopWatch() {
  const [time, setTime]   = useState(0)
  const intervalRef       = useRef<NodeJS.Timeout | null>(null)  // mutable, no re-render

  const start = () => {
    intervalRef.current = setInterval(() => setTime(t => t + 1), 1000)
  }
  const stop = () => {
    if (intervalRef.current) clearInterval(intervalRef.current)
  }

  return (
    <>
      <p>{time}s</p>
      <button onClick={start}>Start</button>
      <button onClick={stop}>Stop</button>
    </>
  )
}
```

## ref vs state

| | `useRef` | `useState` |
|---|---|---|
| Triggers re-render? | ❌ No | ✅ Yes |
| Survives re-renders? | ✅ Yes | ✅ Yes |
| Synchronous access | ✅ Always current | ❌ Batch-delayed |
| Use case | DOM access, timer IDs, previous values | UI state |

## Common Patterns

### Track previous value

```tsx
function usePrevious<T>(value: T) {
  const ref = useRef<T | undefined>(undefined)

  useEffect(() => {
    ref.current = value   // runs AFTER render, stores current as "previous" for next render
  })

  return ref.current   // returns value from previous render
}

// Usage
function Counter() {
  const [count, setCount] = useState(0)
  const prev = usePrevious(count)

  return <p>{prev} → {count}</p>
}
```

### Avoid stale closures in effects

```tsx
function Component({ onSave }: { onSave: (data: Data) => void }) {
  const onSaveRef = useRef(onSave)

  // Keep ref current without causing effect to re-run
  useEffect(() => { onSaveRef.current = onSave })

  useEffect(() => {
    const id = setInterval(() => {
      onSaveRef.current(getData())  // always calls the latest onSave
    }, 5000)
    return () => clearInterval(id)
  }, [])  // no dependency on onSave — won't re-create interval on every render
}
```

### Measure DOM elements

```tsx
function useElementSize() {
  const ref = useRef<HTMLDivElement>(null)
  const [size, setSize] = useState({ width: 0, height: 0 })

  useEffect(() => {
    if (!ref.current) return
    const observer = new ResizeObserver(([entry]) => {
      setSize({ width: entry.contentRect.width, height: entry.contentRect.height })
    })
    observer.observe(ref.current)
    return () => observer.disconnect()
  }, [])

  return { ref, ...size }
}
```

## forwardRef — Expose Refs to Parents

When you need a parent to access a child's DOM node:

```tsx
// Define a component that forwards its ref to an inner DOM element
const FancyInput = forwardRef<HTMLInputElement, { placeholder?: string }>(
  ({ placeholder }, ref) => (
    <div className="input-wrapper">
      <input ref={ref} placeholder={placeholder} className="fancy-input" />
    </div>
  )
)

// Parent accesses the underlying <input>
function Form() {
  const inputRef = useRef<HTMLInputElement>(null)

  return (
    <>
      <FancyInput ref={inputRef} placeholder="Fancy!" />
      <button onClick={() => inputRef.current?.focus()}>Focus</button>
    </>
  )
}
```

## useImperativeHandle — Control Exposed API

Limit what the parent can do with the ref:

```tsx
interface VideoHandle {
  play: () => void
  pause: () => void
  seek: (seconds: number) => void
}

const VideoPlayer = forwardRef<VideoHandle, { src: string }>(({ src }, ref) => {
  const videoRef = useRef<HTMLVideoElement>(null)

  useImperativeHandle(ref, () => ({
    play:  () => videoRef.current?.play(),
    pause: () => videoRef.current?.pause(),
    seek:  (s) => { if (videoRef.current) videoRef.current.currentTime = s },
    // currentTime, volume, etc. are NOT exposed — parent can't access them
  }))

  return <video ref={videoRef} src={src} />
})

// Parent
const playerRef = useRef<VideoHandle>(null)
playerRef.current?.play()   // only the 3 exposed methods work
```

## Summary

<Callout type="success" title="What we learned">
- `useRef` stores a mutable value that persists across renders without causing re-renders
- Primary uses: accessing DOM elements directly and storing timer IDs / previous values
- Use `forwardRef` to pass refs through component wrappers to underlying DOM nodes
- Use `useImperativeHandle` to expose a controlled imperative API to parent components
- Never read `ref.current` during render — it bypasses React's update cycle
</Callout>
