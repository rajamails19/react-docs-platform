# Render Props

The render prop pattern shares stateful logic between components by passing a function as a prop. The component calls that function with its internal state, letting callers control what gets rendered.

```tsx
// Component accepts a function prop — calls it with its state
<DataProvider render={(data) => <Display data={data} />} />

// Or the children-as-function variant:
<MouseTracker>
  {({ x, y }) => <Cursor x={x} y={y} />}
</MouseTracker>
```

## Classic Example: Mouse Position

```tsx
interface MousePosition {
  x: number
  y: number
}

interface MouseTrackerProps {
  render: (pos: MousePosition) => React.ReactNode
  // OR: children: (pos: MousePosition) => React.ReactNode
}

function MouseTracker({ render }: MouseTrackerProps) {
  const [pos, setPos] = useState<MousePosition>({ x: 0, y: 0 })

  return (
    <div
      style={{ height: '300px', border: '1px solid #ccc' }}
      onMouseMove={e => setPos({ x: e.clientX, y: e.clientY })}
    >
      {render(pos)}
    </div>
  )
}

// Callers decide what to show — the Tracker doesn't know or care
<MouseTracker render={({ x, y }) => (
  <p>Mouse is at ({x}, {y})</p>
)} />

<MouseTracker render={({ x, y }) => (
  <img src="/cursor.png" style={{ position: 'absolute', left: x, top: y }} />
)} />
```

## Data Fetching with Render Props

```tsx
interface FetcherProps<T> {
  url:    string
  render: (state: { data: T | null; loading: boolean; error: string | null }) => React.ReactNode
}

function Fetcher<T>({ url, render }: FetcherProps<T>) {
  const [data,    setData]    = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error,   setError]   = useState<string | null>(null)

  useEffect(() => {
    fetch(url)
      .then(r => r.json())
      .then(setData)
      .catch(e => setError(e.message))
      .finally(() => setLoading(false))
  }, [url])

  return <>{render({ data, loading, error })}</>
}

// Usage — caller controls the UI for each state
<Fetcher<User[]>
  url="/api/users"
  render={({ data, loading, error }) => {
    if (loading) return <Spinner />
    if (error)   return <p>Error: {error}</p>
    return <UserTable users={data!} />
  }}
/>
```

## Render Props vs Custom Hooks

```tsx
// Same logic — render prop version
function MouseTracker({ render }: { render: (pos: Position) => React.ReactNode }) {
  const [pos, setPos] = useState({ x: 0, y: 0 })
  return (
    <div onMouseMove={e => setPos({ x: e.clientX, y: e.clientY })}>
      {render(pos)}
    </div>
  )
}

// Same logic — custom hook version (simpler)
function useMousePosition() {
  const [pos, setPos] = useState({ x: 0, y: 0 })

  useEffect(() => {
    const handler = (e: MouseEvent) => setPos({ x: e.clientX, y: e.clientY })
    window.addEventListener('mousemove', handler)
    return () => window.removeEventListener('mousemove', handler)
  }, [])

  return pos
}

// Consumer is much simpler with hooks
function Cursor() {
  const { x, y } = useMousePosition()
  return <p>({x}, {y})</p>
}
```

Custom hooks replaced render props for most use cases. But render props still shine when the component itself provides the DOM context for the logic (like `MouseTracker` needing a specific `<div>`).

## Summary

<Callout type="success" title="What we learned">
- Render props share logic by calling a function prop with internal state — callers control rendering
- Two variants: `render` prop and children-as-function
- Pattern is powerful for logic tied to specific DOM structure (drag, resize, intersection observers)
- Custom hooks are simpler for pure logic sharing without DOM context
- Today: prefer custom hooks. Render props are still valid in libraries and component APIs
</Callout>
