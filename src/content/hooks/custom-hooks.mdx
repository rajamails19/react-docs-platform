# Custom Hooks

Custom hooks are **JavaScript functions whose names start with `use`** that can call other hooks. They let you extract stateful logic into reusable units — the same way you extract components for reusable UI.

<Callout type="info" title="The key insight">
Components share UI. Custom hooks share **logic**. This is React's primary code-reuse mechanism for stateful behavior.
</Callout>

## Building Your First Custom Hook

Extract repetitive logic from components into a hook:

```tsx
// ❌ Before: copy-pasted in 5 components
function UserProfile({ userId }: { userId: string }) {
  const [data, setData]   = useState<User | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError] = useState<string | null>(null)

  useEffect(() => {
    fetch(`/api/users/${userId}`)
      .then(r => r.json())
      .then(d => { setData(d); setLoading(false) })
      .catch(e => { setError(e.message); setLoading(false) })
  }, [userId])

  // ...
}

// ✅ After: extract to custom hook
function useFetch<T>(url: string) {
  const [data, setData]     = useState<T | null>(null)
  const [loading, setLoading] = useState(true)
  const [error, setError]   = useState<string | null>(null)

  useEffect(() => {
    let cancelled = false
    const controller = new AbortController()
    setLoading(true)

    fetch(url, { signal: controller.signal })
      .then(r => { if (!r.ok) throw new Error(`HTTP ${r.status}`); return r.json() })
      .then(d => { if (!cancelled) { setData(d); setLoading(false) } })
      .catch(e => { if (!cancelled && e.name !== 'AbortError') { setError(e.message); setLoading(false) } })

    return () => { cancelled = true; controller.abort() }
  }, [url])

  return { data, loading, error }
}

// Now any component uses it in one line:
const { data: user, loading, error } = useFetch<User>(`/api/users/${userId}`)
```

## Essential Custom Hooks Library

### useLocalStorage

```tsx
function useLocalStorage<T>(key: string, initial: T) {
  const [value, setValue] = useState<T>(() => {
    try {
      const item = window.localStorage.getItem(key)
      return item ? JSON.parse(item) : initial
    } catch {
      return initial
    }
  })

  const set = useCallback((val: T | ((prev: T) => T)) => {
    setValue(prev => {
      const next = typeof val === 'function' ? (val as (p: T) => T)(prev) : val
      try { window.localStorage.setItem(key, JSON.stringify(next)) } catch {}
      return next
    })
  }, [key])

  return [value, set] as const
}

// Usage — persists across page reloads
const [theme, setTheme] = useLocalStorage('theme', 'light')
```

### useDebounce

```tsx
function useDebounce<T>(value: T, delay = 300): T {
  const [debounced, setDebounced] = useState(value)

  useEffect(() => {
    const timer = setTimeout(() => setDebounced(value), delay)
    return () => clearTimeout(timer)
  }, [value, delay])

  return debounced
}

// Usage — debounce search input before hitting API
function SearchBox() {
  const [query, setQuery]     = useState('')
  const debouncedQuery        = useDebounce(query, 400)

  const { data } = useFetch<Result[]>(`/api/search?q=${debouncedQuery}`)

  return (
    <>
      <input value={query} onChange={e => setQuery(e.target.value)} />
      {data?.map(r => <ResultItem key={r.id} result={r} />)}
    </>
  )
}
```

### useToggle

```tsx
function useToggle(initial = false) {
  const [value, setValue] = useState(initial)
  const toggle  = useCallback(() => setValue(v => !v), [])
  const setTrue  = useCallback(() => setValue(true),  [])
  const setFalse = useCallback(() => setValue(false), [])
  return [value, { toggle, setTrue, setFalse }] as const
}

// Usage
const [open, { toggle, setFalse }] = useToggle()
```

### useMediaQuery

```tsx
function useMediaQuery(query: string): boolean {
  const [matches, setMatches] = useState(() => window.matchMedia(query).matches)

  useEffect(() => {
    const mq = window.matchMedia(query)
    setMatches(mq.matches)
    const handler = (e: MediaQueryListEvent) => setMatches(e.matches)
    mq.addEventListener('change', handler)
    return () => mq.removeEventListener('change', handler)
  }, [query])

  return matches
}

// Usage
const isMobile  = useMediaQuery('(max-width: 768px)')
const isDark    = useMediaQuery('(prefers-color-scheme: dark)')
const isReducedMotion = useMediaQuery('(prefers-reduced-motion: reduce)')
```

### useInterval

```tsx
function useInterval(callback: () => void, delay: number | null) {
  const savedCallback = useRef(callback)

  useEffect(() => { savedCallback.current = callback })

  useEffect(() => {
    if (delay === null) return
    const id = setInterval(() => savedCallback.current(), delay)
    return () => clearInterval(id)
  }, [delay])
}

// Usage — set delay to null to pause
const [paused, setPaused] = useState(false)
useInterval(() => setCount(c => c + 1), paused ? null : 1000)
```

### useOnClickOutside

```tsx
function useOnClickOutside(
  ref: RefObject<HTMLElement>,
  handler: (event: MouseEvent | TouchEvent) => void
) {
  useEffect(() => {
    const listener = (e: MouseEvent | TouchEvent) => {
      if (!ref.current || ref.current.contains(e.target as Node)) return
      handler(e)
    }
    document.addEventListener('mousedown', listener)
    document.addEventListener('touchstart', listener)
    return () => {
      document.removeEventListener('mousedown', listener)
      document.removeEventListener('touchstart', listener)
    }
  }, [ref, handler])
}

// Usage — close dropdown when clicking outside
const menuRef = useRef<HTMLDivElement>(null)
useOnClickOutside(menuRef, () => setOpen(false))
```

## Composing Custom Hooks

Custom hooks compose naturally — build complex behavior from simpler hooks:

```tsx
// Combines useFetch + useDebounce + localStorage for a full search feature
function useSearch<T>(endpoint: string) {
  const [recentSearches, setRecentSearches] = useLocalStorage<string[]>('recent-searches', [])
  const [query, setQuery] = useState('')
  const debouncedQuery    = useDebounce(query, 400)

  const { data, loading, error } = useFetch<T[]>(
    debouncedQuery.trim() ? `${endpoint}?q=${encodeURIComponent(debouncedQuery)}` : ''
  )

  const saveSearch = useCallback((term: string) => {
    setRecentSearches(prev => [term, ...prev.filter(s => s !== term)].slice(0, 10))
  }, [setRecentSearches])

  return { query, setQuery, results: data ?? [], loading, error, recentSearches, saveSearch }
}
```

## Rules of Custom Hooks

1. **Must start with `use`** — this tells React (and the linter) it's a hook
2. **Can call other hooks** — useState, useEffect, useCallback, and other custom hooks
3. **Each instance is isolated** — two components using the same hook don't share state
4. **Follow hook rules** — only call at the top level, not in conditionals/loops

```tsx
// ❌ NOT a custom hook — doesn't start with 'use'
function getUser(id: string) { return useFetch(`/api/users/${id}`) }

// ✅ Properly named custom hook
function useUser(id: string) { return useFetch<User>(`/api/users/${id}`) }

// ✅ Two components use the same hook but have separate state
function ProfileA() { const user = useUser('alice') }
function ProfileB() { const user = useUser('bob')   }
// ProfileA's useUser and ProfileB's useUser are completely independent
```

## Summary

<Callout type="success" title="What we learned">
- Custom hooks extract stateful logic into reusable functions
- They must start with `use` and can call other hooks
- Each usage of a hook creates **independent, isolated state**
- Compose custom hooks to build complex features from simple building blocks
- Essential production hooks: `useFetch`, `useLocalStorage`, `useDebounce`, `useMediaQuery`, `useInterval`
</Callout>
