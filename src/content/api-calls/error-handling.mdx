# Error Handling

Production React apps fail gracefully. That means distinguishing network errors from auth errors, showing useful messages to users, and using Error Boundaries to prevent a single component crash from taking down the whole page.

## API Error Types

```tsx
class ApiError extends Error {
  constructor(
    message: string,
    public status: number,
    public code?: string,
  ) {
    super(message)
    this.name = 'ApiError'
  }
}

async function apiFetch<T>(url: string, options?: RequestInit): Promise<T> {
  const res = await fetch(url, options)

  if (!res.ok) {
    const body = await res.json().catch(() => ({}))
    throw new ApiError(
      body.message ?? res.statusText,
      res.status,
      body.code,
    )
  }

  return res.json() as Promise<T>
}

// Usage with typed error handling
async function fetchUser(id: string) {
  try {
    return await apiFetch<User>(`/api/users/${id}`)
  } catch (err) {
    if (err instanceof ApiError) {
      if (err.status === 404) throw new Error('User not found')
      if (err.status === 401) redirect('/login')
      if (err.status >= 500) throw new Error('Server error, try again')
    }
    throw err
  }
}
```

## Inline Error States

```tsx
function UserProfile({ id }: { id: string }) {
  const [state, setState] = useState<{
    status: 'idle' | 'loading' | 'success' | 'error'
    data:   User | null
    error:  string | null
  }>({ status: 'idle', data: null, error: null })

  const load = useCallback(async () => {
    setState(s => ({ ...s, status: 'loading', error: null }))
    try {
      const user = await fetchUser(id)
      setState({ status: 'success', data: user, error: null })
    } catch (err) {
      setState(s => ({
        ...s,
        status: 'error',
        error: err instanceof Error ? err.message : 'Failed to load',
      }))
    }
  }, [id])

  useEffect(() => { load() }, [load])

  if (state.status === 'loading') return <Skeleton />
  if (state.status === 'error') {
    return (
      <div className="error-container">
        <p>{state.error}</p>
        <button onClick={load}>Retry</button>
      </div>
    )
  }
  if (!state.data) return null

  return <div>{state.data.name}</div>
}
```

## React Error Boundaries

Error Boundaries catch errors during rendering. They're class components (React doesn't have a hook version yet):

```tsx
import { Component, ErrorInfo, ReactNode } from 'react'

interface Props {
  children:  ReactNode
  fallback?: ReactNode
}
interface State {
  hasError: boolean
  error:    Error | null
}

class ErrorBoundary extends Component<Props, State> {
  state: State = { hasError: false, error: null }

  static getDerivedStateFromError(error: Error): State {
    return { hasError: true, error }
  }

  componentDidCatch(error: Error, info: ErrorInfo) {
    // Log to Sentry / DataDog
    console.error('Boundary caught:', error, info.componentStack)
  }

  render() {
    if (this.state.hasError) {
      return this.props.fallback ?? (
        <div className="error-boundary">
          <h2>Something went wrong</h2>
          <p>{this.state.error?.message}</p>
          <button onClick={() => this.setState({ hasError: false, error: null })}>
            Try again
          </button>
        </div>
      )
    }
    return this.props.children
  }
}

// Wrap sections of your app
function App() {
  return (
    <ErrorBoundary fallback={<PageError />}>
      <Header />
      <ErrorBoundary fallback={<SidebarError />}>
        <Sidebar />
      </ErrorBoundary>
      <ErrorBoundary fallback={<ContentError />}>
        <MainContent />
      </ErrorBoundary>
    </ErrorBoundary>
  )
}
```

## Retry Pattern with Exponential Backoff

```tsx
async function fetchWithRetry<T>(
  fn: () => Promise<T>,
  retries = 3,
  baseDelay = 500,
): Promise<T> {
  for (let attempt = 0; attempt < retries; attempt++) {
    try {
      return await fn()
    } catch (err) {
      const isLast = attempt === retries - 1
      const isRetryable = !(err instanceof ApiError && err.status < 500)

      if (isLast || !isRetryable) throw err

      // Exponential backoff: 500ms, 1000ms, 2000ms
      await new Promise(r => setTimeout(r, baseDelay * 2 ** attempt))
    }
  }
  throw new Error('Max retries exceeded')
}
```

## Summary

<Callout type="success" title="What we learned">
- Create a typed `ApiError` class to distinguish HTTP status codes in catch blocks
- Model state as `'idle' | 'loading' | 'success' | 'error'` union — prevents impossible states
- Always show a Retry button for transient errors (network, 500s)
- Error Boundaries catch rendering errors and prevent full-app crashes — use at multiple levels
- Retry with exponential backoff for server errors; never retry on 4xx client errors
</Callout>
