# Controlled Inputs

A **controlled input** is a form element whose value is driven entirely by React state. React owns the value; the DOM does not. Every keystroke flows through state, giving you complete control over what the input displays, when it updates, and how it validates.

## The Core Pattern: `value` + `onChange`

```tsx
function NameInput() {
  const [name, setName] = useState('')

  return (
    <input
      type="text"
      value={name}                        // React state → DOM
      onChange={e => setName(e.target.value)} // DOM event → React state
    />
  )
}
```

The data flows in a loop:

<MermaidDiagram chart={`flowchart LR
    S[React State] -->|value prop| I[Input Element]
    I -->|onChange event| H[Event Handler]
    H -->|setState| S`} />

This is called **one-way data binding**. The input can only display what is in state. When the user types, `onChange` fires, which calls `setState`, which re-renders with the new value. Without the `onChange` handler the input would be read-only (React would block keystrokes to maintain the state-value contract).

## Text Input

```tsx
function TextInput() {
  const [value, setValue] = useState('')

  return (
    <div>
      <label htmlFor="username">Username</label>
      <input
        id="username"
        type="text"
        value={value}
        onChange={e => setValue(e.target.value)}
        placeholder="Enter your username"
      />
      <p>Characters: {value.length} / 20</p>
    </div>
  )
}
```

<Callout type="info">
Always pair `<label>` with `htmlFor` (matching the input's `id`). This is required for accessibility — screen readers announce the label when the input receives focus, and clicking the label focuses the input.
</Callout>

## Textarea

`<textarea>` in React is self-closing and accepts a `value` prop, unlike the HTML version which puts its content between tags:

```tsx
function BiographyInput() {
  const [bio, setBio] = useState('')
  const MAX = 200

  return (
    <div>
      <label htmlFor="bio">Biography</label>
      <textarea
        id="bio"
        value={bio}
        onChange={e => setBio(e.target.value.slice(0, MAX))} // enforce max length
        rows={4}
        placeholder="Tell us about yourself…"
      />
      <p style={{ color: bio.length >= MAX ? 'red' : 'inherit' }}>
        {bio.length} / {MAX}
      </p>
    </div>
  )
}
```

## Select Dropdown

Set the selected option by placing the `value` prop on `<select>`, not on `<option>`:

```tsx
const COUNTRIES = ['United States', 'Canada', 'United Kingdom', 'Australia', 'Germany']

function CountrySelect() {
  const [country, setCountry] = useState('')

  return (
    <div>
      <label htmlFor="country">Country</label>
      <select
        id="country"
        value={country}
        onChange={e => setCountry(e.target.value)}
      >
        <option value="">-- Select a country --</option>
        {COUNTRIES.map(c => (
          <option key={c} value={c}>{c}</option>
        ))}
      </select>
    </div>
  )
}
```

For multi-select, use an array in state:

```tsx
function MultiSelect() {
  const [selected, setSelected] = useState<string[]>([])

  const handleChange = (e: React.ChangeEvent<HTMLSelectElement>) => {
    const values = Array.from(e.target.selectedOptions, opt => opt.value)
    setSelected(values)
  }

  return (
    <select multiple value={selected} onChange={handleChange}>
      <option value="react">React</option>
      <option value="vue">Vue</option>
      <option value="angular">Angular</option>
    </select>
  )
}
```

## Checkbox

Checkboxes use `checked` (not `value`) and toggle a boolean:

```tsx
function NewsletterCheckbox() {
  const [subscribed, setSubscribed] = useState(false)

  return (
    <label>
      <input
        type="checkbox"
        checked={subscribed}
        onChange={e => setSubscribed(e.target.checked)}
      />
      {' '}Subscribe to newsletter
    </label>
  )
}
```

For a group of checkboxes mapping to a set of values:

```tsx
const INTERESTS = ['React', 'TypeScript', 'Node.js', 'GraphQL', 'CSS']

function InterestCheckboxes() {
  const [selected, setSelected] = useState<Set<string>>(new Set())

  const toggle = (interest: string) => {
    setSelected(prev => {
      const next = new Set(prev)
      next.has(interest) ? next.delete(interest) : next.add(interest)
      return next
    })
  }

  return (
    <fieldset>
      <legend>Select your interests</legend>
      {INTERESTS.map(interest => (
        <label key={interest} style={{ display: 'block' }}>
          <input
            type="checkbox"
            checked={selected.has(interest)}
            onChange={() => toggle(interest)}
          />
          {' '}{interest}
        </label>
      ))}
      <p>Selected: {[...selected].join(', ') || 'none'}</p>
    </fieldset>
  )
}
```

## Radio Buttons

Radio buttons share a `name` attribute in HTML, but in controlled React you manage the value in state:

```tsx
type Plan = 'free' | 'pro' | 'enterprise'

const PLANS: { value: Plan; label: string; price: string }[] = [
  { value: 'free',       label: 'Free',       price: '$0/mo' },
  { value: 'pro',        label: 'Pro',        price: '$12/mo' },
  { value: 'enterprise', label: 'Enterprise', price: 'Contact us' },
]

function PlanSelector() {
  const [plan, setPlan] = useState<Plan>('free')

  return (
    <fieldset>
      <legend>Choose a plan</legend>
      {PLANS.map(({ value, label, price }) => (
        <label key={value} style={{ display: 'block', marginBottom: 8 }}>
          <input
            type="radio"
            name="plan"          // HTML name groups them for keyboard navigation
            value={value}
            checked={plan === value}
            onChange={() => setPlan(value)}
          />
          {' '}{label} — {price}
        </label>
      ))}
    </fieldset>
  )
}
```

## Handling Multiple Inputs with One Handler

For forms with many fields, maintain a single state object and use the input's `name` attribute as the key:

```tsx
interface SignupForm {
  firstName: string
  lastName: string
  email: string
  password: string
}

function SignupForm() {
  const [form, setForm] = useState<SignupForm>({
    firstName: '',
    lastName: '',
    email: '',
    password: '',
  })

  // Generic handler: reads name and value from the event target
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target
    setForm(prev => ({ ...prev, [name]: value }))
  }

  return (
    <form>
      <input name="firstName" value={form.firstName} onChange={handleChange} placeholder="First name" />
      <input name="lastName"  value={form.lastName}  onChange={handleChange} placeholder="Last name" />
      <input name="email"     value={form.email}     onChange={handleChange} placeholder="Email" type="email" />
      <input name="password"  value={form.password}  onChange={handleChange} placeholder="Password" type="password" />
    </form>
  )
}
```

<Callout type="tip">
The `[name]: value` computed property syntax (`[e.target.name]`) is the key to the single-handler pattern. Because each input's `name` prop matches the corresponding key in the state object, one handler updates any field.
</Callout>

## Controlled Form with Inline Validation

Validate on each change to give users instant feedback:

```tsx
interface RegistrationForm {
  username: string
  email: string
  password: string
}

interface FormErrors {
  username?: string
  email?: string
  password?: string
}

function validate(form: RegistrationForm): FormErrors {
  const errors: FormErrors = {}

  if (form.username.length < 3)
    errors.username = 'Username must be at least 3 characters.'
  if (!/^[^\s@]+@[^\s@]+\.[^\s@]+$/.test(form.email))
    errors.email = 'Please enter a valid email address.'
  if (form.password.length < 8)
    errors.password = 'Password must be at least 8 characters.'

  return errors
}

function RegistrationForm() {
  const [form, setForm] = useState<RegistrationForm>({
    username: '',
    email: '',
    password: '',
  })
  const [touched, setTouched] = useState<Partial<Record<keyof RegistrationForm, boolean>>>({})

  const errors = validate(form)
  const isValid = Object.keys(errors).length === 0

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target
    setForm(prev => ({ ...prev, [name]: value }))
  }

  const handleBlur = (e: React.FocusEvent<HTMLInputElement>) => {
    setTouched(prev => ({ ...prev, [e.target.name]: true }))
  }

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    // Mark all fields as touched to show all errors
    setTouched({ username: true, email: true, password: true })
    if (isValid) {
      console.log('Submit:', form)
    }
  }

  const fieldError = (field: keyof RegistrationForm) =>
    touched[field] ? errors[field] : undefined

  return (
    <form onSubmit={handleSubmit} noValidate>
      <div>
        <label htmlFor="username">Username</label>
        <input
          id="username"
          name="username"
          value={form.username}
          onChange={handleChange}
          onBlur={handleBlur}
          aria-invalid={!!fieldError('username')}
          aria-describedby={fieldError('username') ? 'username-error' : undefined}
        />
        {fieldError('username') && (
          <span id="username-error" role="alert">{fieldError('username')}</span>
        )}
      </div>

      <div>
        <label htmlFor="email">Email</label>
        <input
          id="email"
          name="email"
          type="email"
          value={form.email}
          onChange={handleChange}
          onBlur={handleBlur}
          aria-invalid={!!fieldError('email')}
          aria-describedby={fieldError('email') ? 'email-error' : undefined}
        />
        {fieldError('email') && (
          <span id="email-error" role="alert">{fieldError('email')}</span>
        )}
      </div>

      <div>
        <label htmlFor="password">Password</label>
        <input
          id="password"
          name="password"
          type="password"
          value={form.password}
          onChange={handleChange}
          onBlur={handleBlur}
          aria-invalid={!!fieldError('password')}
          aria-describedby={fieldError('password') ? 'password-error' : undefined}
        />
        {fieldError('password') && (
          <span id="password-error" role="alert">{fieldError('password')}</span>
        )}
      </div>

      <button type="submit">Create Account</button>
    </form>
  )
}
```

## Reset Form

Because state is the source of truth, resetting is trivial:

```tsx
const INITIAL_FORM: SignupForm = { firstName: '', lastName: '', email: '', password: '' }

function ResettableForm() {
  const [form, setForm] = useState(INITIAL_FORM)

  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const { name, value } = e.target
    setForm(prev => ({ ...prev, [name]: value }))
  }

  const resetForm = () => setForm(INITIAL_FORM)

  const handleSubmit = (e: React.FormEvent) => {
    e.preventDefault()
    console.log('Submitting:', form)
    resetForm()
  }

  return (
    <form onSubmit={handleSubmit}>
      {/* ...inputs... */}
      <button type="submit">Submit</button>
      <button type="button" onClick={resetForm}>Reset</button>
    </form>
  )
}
```

<Callout type="info">
The `type="button"` attribute on the Reset button is critical. Without it, a button inside a `<form>` defaults to `type="submit"`, which would trigger form submission when clicked. Always be explicit about button types inside forms.
</Callout>
