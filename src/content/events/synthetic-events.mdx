# Synthetic Events

React wraps native browser events in `SyntheticEvent` objects that normalize behavior across browsers. The API mirrors the native DOM Event API but works identically in every browser.

## What Is a SyntheticEvent?

```tsx
function Form() {
  const handleSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    e.preventDefault()          // same as native DOM event
    e.stopPropagation()         // same as native DOM event
    console.log(e.target)       // the form element
    console.log(e.nativeEvent)  // the underlying browser event
  }

  return <form onSubmit={handleSubmit}>...</form>
}
```

React's event system uses a single event listener at the root (`document`) instead of attaching listeners to individual DOM nodes. This is more memory efficient and avoids memory leaks.

## Typed Event Handlers

TypeScript gives you specific event types for each element:

```tsx
function AllEvents() {
  // Mouse events
  const handleClick    = (e: React.MouseEvent<HTMLButtonElement>) => {}
  const handleMouseMove = (e: React.MouseEvent<HTMLDivElement>) => {}

  // Keyboard events
  const handleKeyDown  = (e: React.KeyboardEvent<HTMLInputElement>) => {}

  // Form events
  const handleSubmit   = (e: React.FormEvent<HTMLFormElement>) => {}
  const handleChange   = (e: React.ChangeEvent<HTMLInputElement>) => {}
  const handleSelect   = (e: React.ChangeEvent<HTMLSelectElement>) => {}

  // Focus events
  const handleFocus    = (e: React.FocusEvent<HTMLInputElement>) => {}
  const handleBlur     = (e: React.FocusEvent<HTMLInputElement>) => {}

  // Drag events
  const handleDrop     = (e: React.DragEvent<HTMLDivElement>) => {}

  // Clipboard events
  const handlePaste    = (e: React.ClipboardEvent<HTMLInputElement>) => {}
}
```

## Reading Event Data

```tsx
function SearchInput() {
  const handleChange = (e: React.ChangeEvent<HTMLInputElement>) => {
    const value = e.target.value       // current input value
    const name  = e.target.name        // input name attribute
    console.log(value, name)
  }

  const handleKeyDown = (e: React.KeyboardEvent<HTMLInputElement>) => {
    if (e.key === 'Enter') {
      console.log('Search submitted')
    }
    if (e.key === 'Escape') {
      e.currentTarget.blur()
    }
    // Modifier keys
    if (e.ctrlKey && e.key === 'a') {
      console.log('Select all')
    }
  }

  return (
    <input
      name="search"
      onChange={handleChange}
      onKeyDown={handleKeyDown}
    />
  )
}
```

## stopPropagation vs preventDefault

```tsx
function EventControl() {
  const handleButtonClick = (e: React.MouseEvent<HTMLButtonElement>) => {
    // Stops event from bubbling up to parent elements
    e.stopPropagation()
    console.log('Button clicked — parent div will NOT hear this')
  }

  const handleFormSubmit = (e: React.FormEvent<HTMLFormElement>) => {
    // Prevents default browser behavior (page reload on form submit)
    e.preventDefault()
    // Handle submission in JavaScript instead
    submitData()
  }

  const handleLinkClick = (e: React.MouseEvent<HTMLAnchorElement>) => {
    // Prevents navigation
    e.preventDefault()
    // Do something custom instead
    showModal()
  }

  return (
    <div onClick={() => console.log('div clicked')}>
      <button onClick={handleButtonClick}>Click me</button>
    </div>
  )
}
```

## Summary

<Callout type="success" title="What we learned">
- `SyntheticEvent` wraps native browser events with a cross-browser normalized API
- Use specific TypeScript types: `React.ChangeEvent<HTMLInputElement>`, `React.MouseEvent<HTMLButtonElement>`, etc.
- `e.preventDefault()` — stop default browser action (form submit, link navigation)
- `e.stopPropagation()` — stop event from bubbling to parent elements
- `e.nativeEvent` — access the underlying browser event if needed
</Callout>
