# Higher-Order Components (HOC)

A Higher-Order Component is a function that takes a component and returns a new component with enhanced behavior. HOCs implement the DRY principle for cross-cutting concerns like authentication, error handling, and analytics.

```tsx
function withEnhancement<P>(WrappedComponent: React.ComponentType<P>) {
  return function EnhancedComponent(props: P) {
    // Add behavior here
    return <WrappedComponent {...props} />
  }
}
```

## Real-World Example: withAuth

```tsx
interface WithAuthProps {
  user: User
}

// HOC that guards a route — redirects to login if not authenticated
function withAuth<P extends WithAuthProps>(
  WrappedComponent: React.ComponentType<P>
) {
  function AuthenticatedComponent(props: Omit<P, keyof WithAuthProps>) {
    const { user, isLoading } = useAuth()

    if (isLoading) return <Spinner />

    if (!user) {
      return <Navigate to="/login" replace />
    }

    return <WrappedComponent {...(props as P)} user={user} />
  }

  // Preserve display name for DevTools
  AuthenticatedComponent.displayName = `withAuth(${WrappedComponent.displayName ?? WrappedComponent.name})`

  return AuthenticatedComponent
}

// Usage
function AdminDashboard({ user }: WithAuthProps) {
  return <h1>Welcome, {user.name}</h1>
}

// Protected version — no auth code in AdminDashboard itself
export const ProtectedAdminDashboard = withAuth(AdminDashboard)
```

## withLoading HOC

```tsx
interface WithLoadingProps {
  isLoading: boolean
}

function withLoading<P extends WithLoadingProps>(
  WrappedComponent: React.ComponentType<P>,
  Fallback: React.ComponentType = Spinner
) {
  return function LoadingComponent(props: P) {
    if (props.isLoading) return <Fallback />
    return <WrappedComponent {...props} />
  }
}

// Combine HOCs
const UserListWithLoading = withLoading(withAuth(UserList))
```

## withErrorBoundary HOC

```tsx
function withErrorBoundary<P extends object>(
  WrappedComponent: React.ComponentType<P>,
  FallbackComponent?: React.ComponentType<{ error: Error }>
) {
  return function SafeComponent(props: P) {
    return (
      <ErrorBoundary fallback={FallbackComponent ? <FallbackComponent error={new Error()} /> : undefined}>
        <WrappedComponent {...props} />
      </ErrorBoundary>
    )
  }
}
```

## HOC vs Custom Hook: When to Use Each

```tsx
// HOC — good for: wrapping with JSX, redirects, conditional rendering
const ProtectedPage = withAuth(Dashboard)

// Custom Hook — good for: sharing logic, no JSX needed
function useProtectedData(url: string) {
  const { user } = useAuth()
  if (!user) throw new Error('Not authenticated')
  return useFetch(url)
}
```

function withTheme<P>(Component: ComponentType<P & { bg: string; textColor: string }>) {
  return function Themed(props: P & { theme: 'light' | 'dark' }) {
    const { theme, ...rest } = props
    const bg        = theme === 'dark' ? '#111827' : '#ffffff'
    const textColor = theme === 'dark' ? '#f9fafb'  : '#111827'
    return <Component {...rest as P} bg={bg} textColor={textColor} />
  }
}

## Summary

<Callout type="success" title="What we learned">
- HOCs are functions that take a component and return an enhanced component
- Use HOCs for wrapping behavior that adds JSX (auth redirects, loading states, error boundaries)
- Always set `displayName` on the returned component for React DevTools
- For sharing logic without adding JSX, prefer Custom Hooks — they're simpler and compose better
- Composing multiple HOCs: `withAuth(withLoading(Component))` — read right to left
</Callout>
